--- title: Suppress Backspace slug: suppress-backspace categories: tags: javascript,dojo date: 2011/02/14 15:57:24 ---

Code to prevent browsers from going back a page on backspace.

    suppressBackspace: function() {
        dojo.connect(dojo.doc, 'onkeypress', function(event) {
            if(event.target.size === undefined &&
               event.target.rows === undefined &&
               event.keyCode == dojo.keys.BACKSPACE) {
                // prevent backspace page nav
                event.preventDefault();
            }
        } );
    },
